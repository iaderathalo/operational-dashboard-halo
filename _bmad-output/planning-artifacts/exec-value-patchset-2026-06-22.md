```ts
    /**
     * 11-1: aggregate risk roll-up over a flat set of stored apps — pure local logic,
     * no fetch. Computed from each node's full descendant set (every metric is a flat
     * count/average over leaf apps, so this equals rolling child aggregates upward).
     * An unmapped app counts against coverage and never inflates the node to a false
     * GREEN ("never a false GREEN", 5-6). Percentages are null for an empty set so an
     * empty node never reads as a fabricated 0%.
     * @param {object[]} applications - descendant applications to aggregate
     * @returns {object} the aggregate roll-up
     */
    private static computeRollup(applications: StoredApplication[]): PortfolioNodeRollup {
        const appCount = applications.length;
        if (appCount === 0) {
            return { appCount: 0, healthyPct: null, coveragePct: null, sloPassingPct: null, avgMaturity: null };
        }
        const pct = (n: number) => Math.round((n / appCount) * 1000) / 10;
        const mapped = applications.filter((a) => a.datadogMapped === true);
        // healthy is over the WHOLE set, not just mapped — an unmapped app is not GREEN.
        const healthy = applications.filter(
            (a) => a.datadogMapped === true && (a.healthStatus || '').toUpperCase() === 'GREEN'
        );
        const sloPassing = applications.filter(
            (a) => a.uptime30d != null && a.slaTarget != null && a.uptime30d >= a.slaTarget
        );
        const maturitySum = applications.reduce(
            (sum, a) => sum + MongoPortfolioRepository.computeMaturity(a).score,
            0
        );
        return {
            appCount,
            healthyPct: pct(healthy.length),
            coveragePct: pct(mapped.length),
            sloPassingPct: pct(sloPassing.length),
            avgMaturity: Math.round((maturitySum / appCount) * 10) / 10,
        };
    }

    /**
     * 11-2: derives the per-app error-budget burn rate from data already synced — no
     * new Datadog call. Window: 30d (matches errorBudgetRemainingPct). Pure logic.
     *
     * burnRate = (100 - uptime30d) / (100 - slaTarget), clamped at 0 when uptime >= target.
     * Bands (Nobl9 / Google SRE): healthy <1.0, fast-burn 1.0–2.0, at-risk >2.0.
     * Honest missing-data (5-6): if uptime30d or slaTarget is missing, or slaTarget is
     * 100 (zero allowed budget => divide-by-zero), return { rate: null, band: 'unknown' }.
     * @param {object} application - stored application document
     * @returns {{rate: number | null, band: string}} burn-rate payload
     */
    private static computeBurnRate(application: StoredApplication) {
        const { uptime30d, slaTarget } = application;
        const allowedBudget = slaTarget != null ? 100 - slaTarget : null;
        if (uptime30d == null || slaTarget == null || allowedBudget == null || allowedBudget <= 0) {
            return { rate: null, band: 'unknown' as const };
        }
        const consumed = Math.max(0, 100 - uptime30d);
        const rate = consumed / allowedBudget;
        const band = rate < 1 ? 'healthy' : rate <= 2 ? 'fast-burn' : 'at-risk';
        return { rate, band: band as 'healthy' | 'fast-burn' | 'at-risk' };
    }

    /**
     * 11-4: derives the executive digest from a scoped set of stored apps. Pure
     * projection, no fetch. Reuses 11-1's computeRollup as the leadership roll-up (no
     * duplicate aggregation). Stamps a freshness summary so a failed last sync (5-8
     * egress blocker) is surfaced rather than presented as current. Top movers need a
     * prior period that is not yet stored, so v1 honestly degrades to a point-in-time
     * snapshot (priorPeriod=null, empty movers) with a note saying so.
     * @param {object[]} applications - scoped applications
     * @param {string} [userEmail] - present when the digest is owner-scoped
     * @returns {object} the digest summary
     */
    private static buildDigest(applications: StoredApplication[], userEmail?: string): DigestSummary {
        const rollup: PortfolioRollup = MongoPortfolioRepository.computeRollup(applications);
        const failed = applications.filter((a) => a.lastSyncStatus === 'error');
        const lastSyncAt =
            applications
                .map((a) => a.lastSyncAt || null)
                .filter((v): v is string => Boolean(v))
                .sort()
                .pop() || null;
        const freshnessOk = failed.length === 0;
        return {
            generatedAt: new Date().toISOString(),
            scope: userEmail ? 'mine' : 'all',
            rollup,
            freshness: {
                ok: freshnessOk,
                failedCount: failed.length,
                lastSyncAt,
                note: freshnessOk
                    ? null
                    : `Last sync failed for ${failed.length} app(s) — values may be stale.`,
            },
            priorPeriod: null,
            movers: [],
            newRisks: [],
            note: 'No prior period to compare against yet — point-in-time snapshot.',
        };
    }
```

    }

            case 'at-risk': return 'metric-bad';
            case 'fast-burn': return 'metric-warn';
            case 'healthy': return 'metric-good';
            default: return 'metric-muted';
        }

        return this.burnRateBandLabels[app.burnRate?.band ?? 'unknown'] ?? 'No SLO data';
    }

    // 11-1 -----------------------------------------------------------------
    /** Roll-up for a node — prefers the API-provided rollup, falls back to a local
     *  derivation so the badges render even if an older payload lacks it. */
    nodeRollup(node: PortfolioNode): PortfolioRollup {
        if (node.rollup) return node.rollup;
        const apps = this.getAllApps(node);
        const n = apps.length;
        if (!n) return { appCount: 0, healthyPct: null, coveragePct: null, sloPassingPct: null, avgMaturity: null };
        const pct = (s: number) => Math.round((s / n) * 1000) / 10;
        return {
            appCount: n,
            healthyPct: pct(apps.filter((a) => a.health === 'green').length),
            coveragePct: pct(apps.filter((a) => a.datadogMapped).length),
            sloPassingPct: null,
            avgMaturity: null,
        };
    }
    rollupPct(value: number | null): string {
        return value != null ? `${value}%` : '—';
    }
    /** Transparent risk ordering (11-1 AC#3): lowest coverage first, then lowest
     *  health, weighted toward larger nodes. Documented = sort key below. */
    private riskScore(r: PortfolioRollup): number {
        const cov = r.coveragePct ?? 0;
        const healthy = r.healthyPct ?? 0;
        return (200 - cov - healthy) * Math.log10(r.appCount + 1);
    }
    topRiskNodes(node: PortfolioNode = this.currentNode, acc: PortfolioNode[] = []): PortfolioNode[] {
        (node.children || []).forEach((c) => { acc.push(c); this.topRiskNodes(c, acc); });
        return acc
            .filter((nd) => this.nodeRollup(nd).appCount > 0)
            .sort((a, b) => this.riskScore(this.nodeRollup(b)) - this.riskScore(this.nodeRollup(a)))
            .slice(0, 5);
    }
    topRiskApps(): PortfolioApp[] {
        return this.getAllApps(this.currentNode)
            .filter((a) => a.health === 'red' || a.health === 'amber' || a.burnRate?.band === 'at-risk')
            .sort((a, b) => this.appRisk(b) - this.appRisk(a))
            .slice(0, 5);

     private static computeMaturity(application: StoredApplication) {
         const {
             datadogMapped,
             monitors,
             uptime30d,
             slaTarget,
             itOwner,
             portfolioOwnerName,
             businessOwner,
         } = application;
         const signals = {
             mapped: datadogMapped ?? false,
             hasMonitor: (monitors?.length ?? 0) > 0,
             hasSLO: uptime30d != null,
             sloPassing: uptime30d != null && slaTarget != null && uptime30d >= slaTarget,
             hasOwner: Boolean(itOwner || portfolioOwnerName || businessOwner),
         };

+                coveragePct: null,
+                sloPassingPct: null,

/**
 * Internal-only endpoint an EXTERNAL scheduler (the same cronjob mechanism that
 * drives the Datadog sync) calls on the configured cadence to run the executive
 * digest (story 11-4). Mirrors {@link InternalSyncController}: exempt from Okta
 * (machine caller) but protected by the shared INTERNAL_SYNC_TOKEN secret. We do
 * NOT add an in-process scheduler — keeping delivery cadence in the platform's
 * existing cron avoids a new runtime dependency and matches how sync is triggered.
 * Resolves to POST /api/v1/internal/digest/run (global prefix in server.ts).
 */
@Controller('internal/digest')
@AllowControllerWithNoBearer()
@UseGuards(InternalSyncGuard)
export default class InternalDigestController {
    /**
     * Creates the internal digest controller.
     * @param {object} dashboardService - service that derives the digest
     * @param {object} logger - Polaris logger instance
     */
    constructor(
        private readonly dashboardService: DashboardService,
        private readonly logger: Logger
    ) {}

    /**
     * Runs the digest over the full (unscoped) portfolio and returns it. Delivery
     * (email/PDF) is performed by the platform mail path / ops layer using the
     * configured DIGEST_RECIPIENTS; this endpoint produces the freshness-honest
     * payload to deliver. See OPEN QUESTION on the delivery channel.
     * @returns {Promise<object>} the generated digest summary
     */
    @Post('run')
    @HttpCode(HttpStatus.OK)
    async run(): Promise<DigestSummary> {
        this.logger.info('Internal executive digest run triggered');
        return this.dashboardService.getDigest();
    }
